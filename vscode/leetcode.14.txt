#include<iostream>
#include<string>
using namespace std;
int main(){
	string nums[4]={"out","ouht","ought","oufh"};
	int m=100;
	for(int i=0;i<4;i++)
	{
		if(m>nums[i].length())
		m=nums[i].length();
			}
	char ch=' ';
	int res=0;
	bool flag=true;
	for(int i=0;i<m;i++)
	{
		ch=nums[0][i];
		for(int j=0;j<m;j++)
		{
			if(ch!=nums[j][i])
			{
			flag=false;
			break;
			}
			
		}
		if(flag==false)
		{
			res=i;
			break;
		}
	}		
	cout<<nums[0].substr(0, res)<<endl; 
}
